Mise à jour plateau

Attributs :
    command : la commande décrite, ici "board_update"
	board : liste des cases
		content : none/pawn/laurel/wall/...
		legion_id : id de la légion (si légionnaire)
		armored : boolean (si légionnaire)

Exemple :
{
    "command": "special_turn",
	"board"
	[
		{
			"content": "none"
		},
		{
			"content": "pawn",
			"legion_id": 1,
			"armored": false
		},
		{
			"content": "laurel"
		}
	]
}
